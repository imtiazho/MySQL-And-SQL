 To Data traverse for single Column(Structure)
SELECT field FROM table_name;

// To Data traverse for single Column
SELECT Name FROM student_details;

// You can traverse multiple filed of a table
SELECT Name, Roll FROM student_details;

// Select whole table use asterisk (*)
SELECT * FROM student_details;

// Find distinct value from a field of table
SELECT DISTINCT City FROM student_details;

// It will show only five data
SELECT * FROM student_details LIMIT 5;

// It skip first 2 data and show 3 data
SELECT * FROM student_details LIMIT 2 3;

// Sort by Name in Ascending
SELECT Name FROM student_details ORDER BY Name;

// Sort by Name in Descending Order
SELECT Name FROM student_details ORDER BY Name DESC;

// You can sort by Based on multiple GPA
SELECT Name, Roll, GPA FROM student_details ORDER BY Name, GPA DESC;

// Operator
SELECT 5 * 3;
SELECT 5 + 3;
SELECT 5 / 3;
SELECT 5 % 3;

// Search data based on certain conditions
SELECT City from student_details WHERE Gender='Female';

// Search, Remove duplicate and sort by City
SELECT DISTINCT City from student_details WHERE Gender='Male' ORDER BY City;

// Search by age
SELECT * from student_details WHERE Age='19';

// Search by comparison operator
SELECT * from student_details WHERE Age < 19;

// Search data with and operator
SELECT Roll, Name, GPA FROM student_details WHERE Roll BETWEEN 102 AND 105;

// Search without != operator
SELECT Roll, Name, GPA FROM student_details WHERE Roll != 103;

// Or Operator 
SELECT * FROM student_details WHERE Gender='Male' OR GPA >= 3.50;

// AND Operator
SELECT * FROM student_details WHERE Gender='Male' AND GPA >= 3.50;

// Nested Condition
SELECT * FROM student_details WHERE City='Lakshmipur' AND (Gender='Male' OR GPA >= 3.50);

// Nested OR
SELECT * FROM `student_details` WHERE City='Dhaka' OR City='Barishal';

// IN Operator
SELECT * FROM `student_details` WHERE City IN ('Dhaka', 'Barishal');

// NOT IN Operator
SELECT * FROM `student_details` WHERE City NOT IN ('Dhaka', 'Barishal');

// LIKE Operator start with ""
SELECT * FROM `student_details` WHERE Name LIKE 'T%';

// LIKE Operator end with ""
SELECT * FROM `student_details` WHERE Name LIKE '%r';

// LIKE Operator include ""
SELECT * FROM `student_details` WHERE Name LIKE '%ba%';

SELECT * FROM `student_details` WHERE Name LIKE '__b%';

SELECT * FROM `student_details` WHERE Name LIKE '%b__';

// As keyword
SELECT Roll as ID, Name as 'First Name' FROM student_details ;

// Crate Table with constraint 
CREATE TABLE teachers (
	ID int NOT NULL AUTO_INCREMENT,
    Name varchar(15) NOT NULL,
    Salay double(10, 2),
    PRIMARY KEY(ID)
);

// Insert data into table
INSERT INTO teachers (ID, Name, Salay) Values(1000, 'Alak', 20000);

// Insert with out ID but it automatic Increment it value by one
INSERT INTO teachers (Name, Salay) Values('Alak', 20000);

// Insert without salary and it will work because we didnt make it NOT NULL
INSERT INTO teachers (Name) Values('Dlak');
