CREATE TABLE student (
    Roll int,
    Name varchar(15),
    Gender varchar(10),
    Age int(5),
    GPA double(3, 2),
    City varchar(15),
    PRIMARY KEY(Roll)
);

RENAME TABLE student TO student_details;

DROP TABLE student_details;

Here is DDL(Data Definition Language) => CREATE, ALTER, DROP 
DML(Data Manipulation Language) => INSERT, SELECT, UPDATE, DELETE


 To Data traverse for single Column(Structure)
SELECT field FROM table_name;

// To Data traverse for single Column
SELECT Name FROM student_details;

// You can traverse multiple filed of a table
SELECT Name, Roll FROM student_details;

// Select whole table use asterisk (*)
SELECT * FROM student_details;

// Find distinct value from a field of table
SELECT DISTINCT City FROM student_details;

// It will show only five data
SELECT * FROM student_details LIMIT 5;

// It skip first 2 data and show 3 data
SELECT * FROM student_details LIMIT 2 3;

// Sort by Name in Ascending
SELECT Name FROM student_details ORDER BY Name;

// Sort by Name in Descending Order
SELECT Name FROM student_details ORDER BY Name DESC;

// You can sort by Based on multiple GPA
SELECT Name, Roll, GPA FROM student_details ORDER BY Name, GPA DESC;
